
```SQL
mysql -u root -p
show databases; // 查看数据库
drop database CoCo; // 删除数据库
create database CoCo; // 创建数据库
use CoCo; // 使用数据库
show tables;
create table users(id int, name varchar(255)); // 创建表
drop table users; // 删除数据库表
insert into users values(1, "Jack"); // 插入新的行
insert into users (name) values ("Bob"); // 指定所要插入数据的列
delete from users where name = "Bob"; // 删除一行
delete from users; 
delete * from users; // 不删除表的情况下删除所有行
```
